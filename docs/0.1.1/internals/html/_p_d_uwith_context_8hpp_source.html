30  /**
31  * \internal
32  *
33  * This class is never instantiated itself, but serves as the base class
34  * for concrete %PDU classes which have a context field (e.g. GetPDU).
35  * For concrete %PDU's without context field, the class PDUwithContext is
36  * used.
37  *
38  * This derives from the PDU class and adds a context field. This field is
39  * only meaningful if PUD::non_default_context is true. If
40  * non_default_context is false, the context will neither be read in when a
41  * PDU is parsed nor will it be serialized by
42  * PDUwithContext::add_header().
43  */
44  class PDUwithContext : public PDU
45  {
46  private:
47  /**
48  * \brief The context
49  */
50  Octet_String context;
51 
52  protected:
53  /**
54  * \brief Parse constructor
55  *
56  * Read the context from a buffer and let the PDU::PDU() parse
57  * constructor do the rest. Is called by the parse constructors of
58  * derived classes. See \ref parsing for details about %PDU
59  * parsing.
60  *
61  * \param pos Iterator pointing to the current stream position. The
62  * iterator is advanced while reading the header.
63  *
64  * \param end Iterator pointing one element past the end of the
65  * current stream. This is needed to mark the end of the
66  * buffer.
67  *
68  * \param big_endian Whether the serialized form of the %PDU is in
69  * big_endian format.
70  *
71  * \exception parse_error If parsing fails, for example because
72  * reading the stream fails or the %PDU is
73  * malformed.
74  */
75  PDUwithContext(binary::const_iterator& pos,
76  const binary::const_iterator& end,
77  bool big_endian)
78  : PDU(pos, end, big_endian)
79  {
80  // read the context (if present)
81  if( non_default_context )
82  {
83  this->context = Octet_String(pos, end, big_endian);
84  }
85  }
86 
87  /**
88  * \brief Add PDU header and context field to the payload
89  *
90  * Add the PDU header and the context field to the payload. Called
91  * by derived classes during serialization.
92  *
93  * \warning The payload must not grow or shrink after a call to
94  * this function as its size is encoded into the header.
95  * \warning The function must not be called twice for the same PDU,
96  * because this would add two headers and thus generate a
97  * malformed PDU.
98  *
99  * Header and context are encoded in big endian format.
100  *
101  * \param type The PDU type, according to RFC 2741, 6.1. "AgentX
102  * PDU Header".
103  *
104  * \param payload The payload of the PDU. The header is added to
105  * the payload, i.e. payload is altered by this
106  * function.
107  */
108  void add_header(type_t type, binary& payload) const
109  {
110  // Prepend context to payload, if present
111  if( non_default_context )
112  {
113  payload.insert(0, context.serialize());
114  }
115 
116  // Add header
117  PDU::add_header(type, payload);
118  }
119 
120  /**
121  * \brief Default Constructor
122  *
123  * Sets the state of the object to the defaults as set by the
124  * PDU::PDU() constructor. The non_default_context flag is set to
125  * false by default, which means that the object has no context.
126  */
127  PDUwithContext() {}
128 
129  public:
130  /**
131  * \brief Whether the PDU has a context
132  */
133  bool has_context()
134  {
135  return this->non_default_context;
136  }
137 
138  /**
139  * \brief Get context
140  *
141  * \return The internally stored context.
142  *
143  * \warning The returned context is only valid if the PDU actually
144  * has a context. This can be determined using
145  * has_context().
146  */
147  Octet_String get_context()
148  {
149  return this->context;
150  }
151 
152  /**
153  * \brief Set the PDU's context
154  *
155  * \param value The new context. The given value may be the empty
156  * string, which does \em not delete the context, but
157  * set the empty string as context. To remove the
158  * context from the PDU, use remove_context().
159  */
160  void set_context(Octet_String value)
161  {
162  this->context = value;
163  this->non_default_context = true;
164  }
165 
166  /**
167  * \brief Removes the context from the PDU.
168  *
169  * Afterwards the PDU has no context any longer.
170  */
171  void remove_context()
172  {
173  this->non_default_context = false;
174  }
175 
176  };
177 }

00063 PDU::PDU(data_t::const_iterator& pos,
00064          const data_t::const_iterator& end,
00065          bool big_endian)
00066 {

00071

00095     // skip payload length (not needed here)
00096     pos += 4;

00106 
00107     // check protocol version
00108     byte_t version = buf[0];

00110     {
00111         // Wrong protocol:
00112         // throw exception
00113         throw( version_error() );

00191 void PDU::add_header(type_t type, data_t& payload) const
00192 {
00193     /* Construct header */
00194     data_t header;
00195 
00196     // Protocol version
00197     header.push_back(1);
00198 
00199     // Type
00200     header.push_back(type);
00201 
00202     // flags
00203     byte_t flags = 0;
00204     if(instance_registration) flags |= (1<<0);
00205     if(new_index)             flags |= (1<<1);
00206     if(any_index)             flags |= (1<<2);
00207     if(non_default_context)   flags |= (1<<3);
00208                               flags |= (1<<4);  // We always use big endian
00209     header.push_back(flags);
00210 
00211     // reserved field
00212     header.push_back(0);
00213 
00214     // remaining fields
00215     write32(header, sessionID);
00216     write32(header, transactionID);
00217     write32(header, packetID);
00218     write32(header, payload.size());    // payload length
00219 
00220     // Add header to payload
00221     payload.insert(0, header);
00222 }

00701 static void callback_for_response(const boost::system::error_code& result,

00715 /*
00716  * Timeout handling:
00717  * 
00718  * The timeout handling is realized using a boost::asio::deadline_timer 
00719  * object, which is set up to call the callback_for_response() function when 
00720  * it expires (or when it fails). The variable 'timer_result' is intially set 
00721  * to 'in_progress'.  The callback will set it to 'success' or 'fail' when the 
00722  * timer expires respectively fails.
00723  *
00724  * Then the io_service.run_one() function is invoked repeatedly until either 
00725  * the timer expired or the desired ResponsePDU arrived. Note that 
00726  * io_service.run_one() may service other asynchronous operations first, e.g.  
00727  * a get request.
00728  *
00729  * Finally either the received ResponsePDU is returned or a timeout exception 
00730  * is thrown.
00731  */
00732 boost::shared_ptr<ResponsePDU>
00733 connector::wait_for_response(uint32_t packetID)
00734 {
00735     if(this->socket == 0)
00736     {
00737         // We are currently not conected
00738         throw disconnected();
00739     }
00740 
00741     // Indicate that we are waiting for a specific response:
00742     // We add a null pointer to the map
00743     responses[packetID] = boost::shared_ptr<ResponsePDU>();
00744 
00745     // Start timeout timer
00746     boost::asio::deadline_timer timer(*(this->io_service));
00747     status_t timer_result = in_progress; // callback stores result here
00748     timer.expires_from_now( boost::posix_time::seconds(this->timeout) );
00749     timer.async_wait( boost::bind(callback_for_response,
00750                                   boost::asio::placeholders::error,
00751                                   &timer_result) );
00752 
00753     // process asio events until ResponsePDU arrives or timeout expires
00754     try
00755     {
00756         while(   this->responses[packetID].get() == 0 // no response was stored
00757                  && timer_result == in_progress)
00758         {
00759             this->io_service->run_one();
00760         }
00761     }
00762     catch(boost::system::system_error)
00763     {
00764         // run_one() failed
00765         // -> disconnect and fail
00766         this->disconnect();
00767         throw disconnected();
00768     }
00769 
00770     // Check the result
00771     if( this->responses[packetID].get() != 0 )
00772     {
00773         // ResponsePDU arrived:
00774         // 1. Cancel timer
00775         // 2. Erase response from map
00776         // 3. Return response
00777         timer.cancel();
00778         shared_ptr<ResponsePDU> retval = this->responses[packetID];
00779         this->responses.erase( packetID );
00780         return retval;
00781     }
00782     else
00783     {
00784         // Timer expired or failed before ResponsePDU arrived
00785         throw(timeout_error());
00786     }
00787 }

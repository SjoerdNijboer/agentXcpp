63  /**
64  * \brief Result type for the TestSet validation.
65  *
66  * A value of this type is returned during a TestSet operation
67  * according to RFC 2741, 7.2.4.1. "Subagent Processing of the
68  * agentx-TestSet-PDU".
69  *
70  * \note The value 'wrongEncoding' is not used within the
71  * agentXcpp implementation and is therefore not part of this
72  * enumeration.
73  *
74  * \internal
75  *
76  * The numeric values are given in RFC 2741, 7.2.4.1. "Subagent
77  * Processing of the agentx-TestSet-PDU" and must be in sync with
78  * the corresponding errors defined in ResponsePDU::error_t.
79  */
80  enum testset_result_t
81  {
82  /**
83  * Validation succeeded.
84  */
85  noError = 0,
86 
87  /**
88  * The variable could not be set, and there is no other
89  * enumeration value to express the reason.
90  */
91  genErr = 5,
92 
93  /**
94  * Access denied (i.e. the object is read-only).
95  */
96  noAccess = 6,
97 
98  /**
99  * The object cannot be set to the requested type because it
100  * has the wrong type (e.g. an OctetString variable cannot be
101  * set to an Integer value).
102  */
103  wrongType = 7,
104 
105  /**
106  * The requested new value has the wrong length. E.g. an
107  * OctetString variable may be restricted to 8 characters, but
108  * the requested new string has 15 characters.
109  */
110  wrongLength = 8,
111 
112  /**
113  * The requested value could under no circumstances be assigned
114  * to the variable (although the variable is read-write). For
115  * example, certain values may be disallowed for a specific
116  * Integer variable.
117  */
118  wrongValue = 10,
119 
120  /**
121  * The variable does not exist and could not ever be created.
122  */
123  noCreation = 11,
124 
125  /**
126  * The requested value could under other circumstances be held
127  * by the variable, but is presently inconsistent or otherwise
128  * unable to be assigned to the variable.
129  */
130  inconsistentValue = 12,
131 
132  /**
133  * A resource which is required to perform the actual Set
134  * operation is unavailable. A Set operation would fail because

136  */
137  resourceUnavailable = 13,

139  /**

145  notWritable = 17,

155  /**
156  * \brief Validate whether a Set operation would be successful.
157  *
158  * This method is called when a TestSet request is received. It
159  * shall check whether a Set operation is possible for the
160  * variable. It shall acquire the resources needed to perform the
161  * Set operation (but the Set shall not yet performed).
162  *
163  * \note This is the only method which receives the new value to be
164  * set. An implementation must save the new value for
165  * subsequent operations (i.e. handle_commitset()).
166  *
167  * \return The result of the validation.
168  */
169  virtual testset_result_t handle_testset(shared_ptr<AbstractValue>) = 0;
170 
171  /**
172  * \brief Release resources after a Set operation.
173  *
174  * This method is called when a CleanupSet request is received. It
175  * shall release all resources previously allocated by
176  * handle_testset() (if any). If no resources were allocated, this
177  * method is not required to do anything.
178  */
179  virtual void handle_cleanupset() = 0;
180 
181  /**
182  * \brief Actually perform the Set operation.
183  *
184  * This method is called when a CommitSet request is received for
185  * the variable. It shall perform the Set operation. It shall
186  * report whether the operation succeeded.
187  *
188  * \note The new value is given to handle_testset() prior to
189  * calling handle_commitset().
190  *
191  * \return True on success, false otherwise.
192  */
193  virtual bool handle_commitset()= 0;
194 
195  /**
196  * \brief Undo a Set operation which was already performed.
197  *
198  * This method is called when an UndoSet request is received. It
199  * shall undo the operation performed by handle_commitset().
200  *
201  * \return True on success, false otherwise.
202  */
203  virtual bool handle_undoset() = 0;

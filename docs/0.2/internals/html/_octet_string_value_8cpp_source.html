33 void OctetStringValue::set_value(std::string v)
34 {
35  // Here we convert initial value to a binary string. We do this in three
36  // steps:
37  // 1. get the bare data: v.data()
38  // 2. cast the data to the value type of binary
39  // 3. calculate the size of the data
40  // - v.size() gives us the number of characters
41  // - sizeof(binary::value_type) gives us the size of an character
42  // 4. Assign v to value, giving it the bare data and its size
43  //
44  // This seems to be goofy, but it ensures that our code works even on a
45  // machine where a char is not 8 bit wide, i.e. when char has another size
46  // than uint8_t.
47  value.assign(
48  reinterpret_cast<const binary::value_type*>( v.data() ),
49  v.size() * sizeof( binary::value_type )
50  );
51 }

75 binary OctetStringValue::serialize() const
76 {
77  binary serialized;
78 
79  // encode size (big endian)
80  write32(serialized, value.size());
81 
82  // encode value
83  serialized += value;
84 
85  // Padding bytes
86  int padsize = 4 - (value.size() % 4);
87  if( padsize == 4 ) padsize = 0; // avoid adding 4 padding bytes
88  while( padsize-- )
89  {
90  serialized.push_back(0);
91  }
92 
93  return serialized;
94 }

97 OctetStringValue::OctetStringValue(binary::const_iterator& pos,
98  const binary::const_iterator& end,
99  bool big_endian)
100 {
101  int size;
102 
103  // We need 4 bytes for the size
104  if(end - pos < 4)
105  {
106  throw(parse_error());
107  }
108 
109  // Get size
110  size = read32(pos, big_endian);
111 
112  // Octet String emtpy?
113  if( size == 0 )
114  {
115  // nothing left to parse
116  return;
117  }
118 
119  // We want to read (size) more bytes
120  if((end - pos) < static_cast<binary::iterator::difference_type>(size))
121  {
122  throw(parse_error());
123  }
124 
125  // Get value
126  value.assign(pos, pos+size);
127  pos += size;
128 
129  // Eat padding bytes
130  int padsize = 4 - (size % 4);
131  if( padsize == 4 ) padsize = 0;
132  if(end - pos < padsize)
133  {
134  throw(parse_error());
135  }
136  while( padsize-- )
137  {
138  pos++;
139  }
140 }

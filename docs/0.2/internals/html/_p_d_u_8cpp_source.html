63 PDU::PDU(binary::const_iterator& pos,
64  const binary::const_iterator& end,
65  bool big_endian)
66 {
67  if(end - pos < 20)
68  {
69  throw(parse_error());
70  }
71 
72  // skip protocol version (already checked by get_pdu()) and
73  // type field (not needed here)
74  pos += 2;
75 
76  // read flags
77  uint8_t flags = *pos++;
78  instance_registration = ( flags & (1<<0) ) ? true : false;
79  new_index = ( flags & (1<<1) ) ? true : false;
80  any_index = ( flags & (1<<2) ) ? true : false;
81  non_default_context = ( flags & (1<<3) ) ? true : false;
82 
83  // skip reserved field
84  pos++;
85 
86  // read sessionID
87  sessionID = read32(pos, big_endian);
88 
89  // read transactionID
90  transactionID = read32(pos, big_endian);
91 
92  // read packetID
93  packetID = read32(pos, big_endian);
94 
95  // skip payload length (not needed here)
96  pos += 4;
97 }

102 shared_ptr<PDU> PDU::parse_pdu(binary buf)
103 {
104  // needed for parsing
105  binary::const_iterator pos;
106 
107  // check protocol version
108  uint8_t version = buf[0];
109  if( version != 1 )
110  {
111  // Wrong protocol:
112  // throw exception
113  throw( version_error() );
114  }
115 
116  // read endianess flag
117  uint8_t flags = buf[2];
118  bool big_endian = ( flags & (1<<4) ) ? true : false;
119 
120  // read payload length
121  uint32_t payload_length;
122  pos = buf.begin() + 16;
123  payload_length = read32(pos, big_endian);
124  if( payload_length % 4 != 0 )
125  {
126  // payload length must be a multiple of 4!
127  // See RFC 2741, 6.1. "AgentX PDU Header"
128  // -> throw exception
129  throw( parse_error() );
130  }
131 
132  // read PDU type
133  uint8_t type = buf[1];
134 
135  // create PDU (TODO: complete the list!)
136  shared_ptr<PDU> pdu;
137  pos = buf.begin();
138  const binary::const_iterator end = buf.end();
139  switch(type)
140  {
141  case agentxOpenPDU:
142  pdu.reset(new OpenPDU(pos, end, big_endian));
143  break;
144  case agentxClosePDU:
145  pdu.reset(new ClosePDU(pos, end, big_endian));
146  break;
147  case agentxRegisterPDU:
148  pdu.reset(new RegisterPDU(pos, end, big_endian));
149  break;
150  case agentxUnregisterPDU:
151  pdu.reset(new UnregisterPDU(pos, end, big_endian));
152  break;
153  case agentxResponsePDU:
154  pdu.reset(new ResponsePDU(pos, end, big_endian));
155  break;
156  case agentxCommitSetPDU:
157  pdu.reset(new CommitSetPDU(pos, end, big_endian));
158  break;
159  case agentxUndoSetPDU:
160  pdu.reset(new UndoSetPDU(pos, end, big_endian));
161  break;
162  case agentxTestSetPDU:
163  pdu.reset(new TestSetPDU(pos, end, big_endian));
164  break;
165  case agentxCleanupSetPDU:
166  pdu.reset(new CleanupSetPDU(pos, end, big_endian));
167  break;
168  case agentxGetPDU:
169  pdu.reset(new GetPDU(pos, end, big_endian));
170  break;
171  case agentxGetNextPDU:
172  pdu.reset(new GetNextPDU(pos, end, big_endian));
173  break;
174  case agentxGetBulkPDU:
175  pdu.reset(new GetBulkPDU(pos, end, big_endian));
176  break;
177  case agentxNotifyPDU:
178  pdu.reset(new NotifyPDU(pos, end, big_endian));
179  break;
180  default:
181  // type is invalid
182  throw(parse_error());
183  }
184 
185  // return created PDU
186  return pdu;
187 }

191 void PDU::add_header(type_t type, binary& payload) const
192 {
193  /* Construct header */
194  binary header;
195 
196  // Protocol version
197  header.push_back(1);
198 
199  // Type
200  header.push_back(type);
201 
202  // flags
203  uint8_t flags = 0;
204  if(instance_registration) flags |= (1<<0);
205  if(new_index) flags |= (1<<1);
206  if(any_index) flags |= (1<<2);
207  if(non_default_context) flags |= (1<<3);
208  flags |= (1<<4); // We always use big endian
209  header.push_back(flags);
210 
211  // reserved field
212  header.push_back(0);
213 
214  // remaining fields
215  write32(header, sessionID);
216  write32(header, transactionID);
217  write32(header, packetID);
218  write32(header, payload.size()); // payload length
219 
220  // Add header to payload
221  payload.insert(0, header);
222 }

51 void UnixDomainConnector::do_connect()
52 {
53  // Inspect socket state
54  switch(m_socket.state())
55  {
56  case QLocalSocket::UnconnectedState:
57  // Currently not connected. Start connecting:
58  m_socket.connectToServer(m_filename);
59  break;
60  case QLocalSocket::ConnectedState:
61  // Already connected. Wake connect() and return.
62  m_connection_waitcondition.wakeAll();
63  return;
64  case QLocalSocket::ConnectingState:
65  // Seems that last attempt for connection timed out
66  // and is still unfinished. We will wait again.
67  break;
68  case QLocalSocket::ClosingState:
69  // Last disconnect seems still unfinished.
70  // Wake connect() and return.
71  m_connection_waitcondition.wakeAll();
72  return;
73  }
74 
75  // Protect m_is_connected
76  QMutexLocker locker(&m_mutex_is_connected);
77 
78  // Wait for connection establishment
79  if(m_socket.waitForConnected(m_timeout))
80  {
81  // OK: Set state to 'connected' and wake connect()
82  m_is_connected = true;
83  m_connection_waitcondition.wakeAll();
84  }
85  else
86  {
87  // Error: Set state to 'disconnected' and wake connect()
88  m_is_connected = false;
89  m_connection_waitcondition.wakeAll();
90  }
91 }

153 void UnixDomainConnector::do_receive()
154 {
155  // If a header was read from the socket, but the payload did not
156  // yet arrive completely, the read header is stored here until
157  // more data arrived:
158  static binary last_header;
159 
160  // Read all PDUs into distinct buffers
161  std::list<binary> queue;
162  do
163  {
164  binary buf;
165 
166  // Read header
167  if(last_header.size() != 0)
168  {
169  // Last time, we read a header, but no payload yet.
170  // Therefore we don't read the header here, but continue
171  // with reading the payload.
172  buf = last_header;
173  last_header.clear();
174  }
175  else
176  {
177  // No header read last time.
178  char header[20];
179  if(m_socket.read(header, 20) != 20)
180  {
181  disconnect(); // error!
182  break; // Stop reading PDUs
183  }
184  buf.append(reinterpret_cast<uint8_t*>(header), 20);
185  }
186 
187  // Extract endianness flag
188  bool big_endian = ( buf[2] & (1<<4) ) ? true : false;
189 
190  // Extract payload length
191  uint32_t payload_length;
192  binary::const_iterator pos = buf.begin() + 16;
193  payload_length = read32(pos, big_endian);
194  if( payload_length % 4 != 0 )
195  {
196  // payload length must be a multiple of 4!
197  // See RFC 2741, 6.1. "AgentX PDU Header"
198  // We don't know where next PDU starts within the byte stream,
199  // therefore we disconnect.
200  disconnect(); // error!
201  // stop reading PDU's (but process the ones we got so far)
202  break;
203  }
204 
205  // Read payload
206  if(m_socket.bytesAvailable() < payload_length)
207  {
208  // Payload did not completely arrive. We store the header until
209  // more data arrived:
210  last_header = buf;
211  break;
212  }
213  QScopedArrayPointer<char> payload(new char[payload_length]);
214  qint64 bytes_read = m_socket.read(payload.data(), payload_length);
215  if(bytes_read != payload_length)
216  {
217  disconnect();
218  return;
219  }
220  buf.append(reinterpret_cast<uint8_t*>(payload.data()), payload_length);
221 
222  queue.push_back(buf);
223  }
224  while(m_socket.bytesAvailable() >= 20); // still enough data for next header
225 
226  // Process all received PDU's
227  for(list<binary>::const_iterator i = queue.begin(); i != queue.end(); i++)
228  {
229  // Parse PDU
230  shared_ptr<PDU> pdu;
231  try
232  {
233  pdu = PDU::parse_pdu(*i);
234  }
235  catch(version_error)
236  {
237  return;
238  }
239  catch(parse_error)
240  {
241  return;
242  }
243 
244  // Special case: ResponsePDU's
245  shared_ptr<ResponsePDU> response;
246  response = boost::dynamic_pointer_cast<ResponsePDU>(pdu);
247  if(response)
248  {
249  m_response_mutex.lock();
250  // Was a response
251  std::map< uint32_t, boost::shared_ptr<ResponsePDU> >::iterator i;
252  i = this->m_responses.find( response->get_packetID() );
253  if(i != this->m_responses.end())
254  {
255  // Someone is waiting for this response
256  i->second = response;
257  m_response_mutex.unlock();
258  m_response_arrived.wakeAll();
259  }
260  else
261  {
262  // Nobody was waiting for the response
263  // -> ignore it
264  m_response_mutex.unlock();
265  }
266  }
267  else
268  {
269  // Was not a Response
270  // -> emit signal
271  emit pduArrived(pdu);
272  }
273  }
274 }

276 boost::shared_ptr<ResponsePDU> UnixDomainConnector::request(boost::shared_ptr<PDU> pdu)
277 {
278  m_response_mutex.lock();
279  m_responses[pdu->get_packetID()] = shared_ptr<ResponsePDU>();
280  QMetaObject::invokeMethod(this, "do_send", Q_ARG(boost::shared_ptr<PDU>, pdu));
281 
282  std::map<uint32_t, boost::shared_ptr<ResponsePDU> >::iterator i;
283  do
284  {
285  m_response_arrived.wait(&m_response_mutex);
286  }
287  while ( ! (m_responses[pdu->get_packetID()]) );
288 
289  shared_ptr<ResponsePDU> response = m_responses[pdu->get_packetID()];
290  m_responses.erase(m_responses.find(pdu->get_packetID()));
291  m_response_mutex.unlock();
292 
293  return response;

70  /**
71  * \brief Result type for the TestSet validation.
72  *
73  * A value of this type is returned during a TestSet operation
74  * according to RFC 2741, 7.2.4.1. "Subagent Processing of the
75  * agentx-TestSet-PDU".
76  *
77  * \note The value 'wrongEncoding' is not used within the
78  * agentXcpp implementation and is therefore not part of this
79  * enumeration.
80  *
81  * The numeric values are given in RFC 2741, 7.2.4.1. "Subagent
82  * Processing of the agentx-TestSet-PDU".
83  *
84  * \internal
85  *
86  * \note These values must be in sync with the corresponding
87  * errors defined in agentxcpp::ResponsePDU::error_t.
88  *
89  */
90  enum testset_result_t

93  * Validation succeeded.
94  */

97  /**

99  * enumeration value to express the reason.
100  */
101  genErr = 5,
102 
103  /**
104  * Access denied (i.e. the object is read-only).
105  */
106  noAccess = 6,
107 
108  /**

112  */
113  wrongType = 7,
114

119  */
120  wrongLength = 8,
121 
122  /**

128  wrongValue = 10,

133  noCreation = 11,

165  /**
166  * \internal
167  *
168  * \brief Validate whether a Set operation would be successful.
169  *
170  * This method is called when the SNMP request "TestSet" is
171  * received. It shall check whether a Set operation is possible for
172  * the variable. It shall acquire the resources needed to perform
173  * the Set operation (but the Set shall not yet be performed).
174  *
175  * \note This is the only method which receives the new value to be
176  * set. An implementation must save the new value for
177  * subsequent operations (i.e. handle_commitset()).
178  *
179  * \return The result of the validation.
180  *
181  * \exception The function shall not throw.
182  */
183  virtual testset_result_t handle_testset(QSharedPointer<AbstractVariable>) = 0;

26 GetPDU::GetPDU(binary::const_iterator& pos,
27  const binary::const_iterator& end,
28  bool big_endian)
29  : PDUwithContext(pos, end, big_endian)
30 {
31  // Get SearchRanges until the PDU is completely parsed
32  while( pos < end )
33  {
34  // read starting OidVariable
35  sr.push_back(OidVariable(pos, end, big_endian).value());
36 
37  // read and forget ending OidVariable (but check its include field)
38  if(OidVariable(pos, end, big_endian).value().include() == true)
39  {
40  // Parse error according to RFC 2741, 5.2 "SearchRange":
41  // include field of ending OID must be 0
42  throw( parse_error() );
43  }
44 
45  pos += 4; // ignore empty "end" OidVariable
46  }
47 }

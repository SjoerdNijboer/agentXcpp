31  /**
32  * \brief This class represents an Integer variable as described in RFC
33  * 2741.
34  *
35  * The page \ref variables explains how to use variables.
36  *
37  * \note When an IntegerVariable is used as INDEX within a table, it should
38  * never hold the value 0. This is because the value is then used as
39  * part of an OID, where 0 is forbidden (or at least discouraged).
40  */
41  class IntegerVariable : public AbstractVariable
42  {
43  protected:
44  /**
45  * \brief The current value.
46  *
47  * This is the value which is sent to the master agent on SNMP Get
48  * requests. According to RFC 2578, INTEGER is a signed 32-bit
49  * number. The value can be obtained with value() and set with
50  * setValue().
51  *
52  * The perform_get() method should update this member.
53  *
54  * The methods perform_testset(), perform_commitset(),
55  * perform_cleanupset() and/or perform_undoset() may update this
56  * member, but it is not needed for proper function.
57  */
58  qint32 v;

63  * \brief The new value for the variable. Used during a Set

65  *
66  * The Set operation is performed in up to four steps (TestSet,
67  * CommitSet, CleanupSet, UndoSet). Only the TestSet step
68  * actually
69  * receives the new value. This value is stored here so that it can
70  * be delivered to perform_commitset(), perform_undoset() and perform_cleanupset().
71  */

123  /**
124  * \brief Convert the value to an OID.
125  *
126  * The conversion is done according to RFC 2578,
127  * 7.7. "Mapping of the INDEX clause". The value is
128  * converted to an Oid with a single subid. Note that
129  * INTEGER values are signed, while subids are not.
130  * A negative value with be converted to a big unsigned
131  * subid.
132  *
133  * This method should not be overridden.
134  *
135  * \note If an INTEGER is used in an INDEX clause, the
136  * value 0 should be avoided according to
137  * RFC 2578, 7.7. "Mapping of the INDEX clause".
138  */
139  virtual Oid toOid() const

189  /**
190  * \internal
191  *
192  * \brief Handle a TestSet request.
193  *
194  * This function converts the argument, stores it in \ref new_value
195  * and then calls perform_testset() with the internal value of \ref
196  * new_value. If conversion fails,
197  * \agentxcpp{AbstractVariable::wrongType} is returned and
198  * perform_testset() is not called.
199  *
200  * \param _v The new value for the variable.
201  *
202  * \return \agentxcpp{AbstractVariable::wrongType}
203  * if the conversion fails. Otherwise, the result of
204  * perform_testset() is returned.
205  */
206  virtual testset_result_t handle_testset(QSharedPointer<AbstractVariable> _v)
207  {
208  new_value = qSharedPointerDynamicCast<IntegerVariable>(_v);
209  if (new_value)
210  {
211  // Type matches variable
212  return perform_testset(new_value->value());
213  }
214  else
215  {
216  // Wrong type
217  return wrongType;
218  }

240  * agent).

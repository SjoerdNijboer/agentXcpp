209 void MasterProxy::do_registration(QSharedPointer<RegisterPDU> pdu)
210 {
211  // Are we connected?
212 // if( ! is_connected())
213 // {
214 // throw(disconnected());
215 // }
216 
217  // Send RegisterPDU
218  // (forward exceptions timeout_error and disconnected)
219  QSharedPointer<ResponsePDU> response;
220  response = this->connection->request(pdu);
221 
222 // // Wait for response
223 // // (forward exceptions timeout_error and disconnected)
224 // response = this->connection->wait_for_response(pdu->get_packetID());
225 
226  // Check Response
227  switch(response->get_error())
228  {
229  // General errors:
230 
231  case ResponsePDU::parseError:
232  // Oops, we sent a malformed PDU to the master
233  throw(internal_error());
234 
235  case ResponsePDU::notOpen:
236  // We checked the connection state before, but maybe we lost the
237  // connection during communication...
238  throw(disconnected());
239 
240  case ResponsePDU::unsupportedContext:
241  // We do currently not really support contexts in this library. An
242  // invalid context is thus probably an agentxcpp bug.
243  throw(internal_error());
244 
245  case ResponsePDU::processingError:
246  // master was unable to process the request
247  throw(master_is_unable());
248 
249  case ResponsePDU::noAgentXError:
250  // Hey, it worked!
251  break;
252 
253  // Register-specific errors:
254 
255  case ResponsePDU::duplicateRegistration:
256  throw(duplicate_registration());
257 
258  case ResponsePDU::requestDenied:
259  throw(master_is_unwilling());
260 
261  default:
262  // This is a case of can-not-happen. Probably the master is buggy.
263  // The agentxcpp library is bug-free of course ;-)
264  // We throw a parse error meanwhile, because we didn't expect the
265  // response to look like that...
266  throw(parse_error());
267  }
268 
269  // Finish
270  return;
271 }

313  quint8 priority)
314 {

317 
318  // Remove the registration from registrations list

322  {
323  if( (*r)->get_priority() == priority

329

440 void MasterProxy::handle_getpdu(QSharedPointer<ResponsePDU> response, QSharedPointer<GetPDU> get_pdu)
441 {
442  // Handling according to
443  // RFC 2741, 7.2.3.1 "Subagent Processing of the agentx-Get-PDU"
444 
445  // Extract searchRange list
446  vector<Oid> sr = get_pdu->get_sr();
447 
448  // Iterate over list and handle each Oid separately
449  vector<Oid>::const_iterator i;
450  quint16 index = 1; // Index is 1-based (RFC 2741,
451  // 5.4. "Value Representation"):
452  for(i = sr.begin(); i != sr.end(); i++)
453  {
454  // The name
455  const Oid& name = *i;
456 
457  // Find variable for current OID
458  map< Oid, QSharedPointer<AbstractVariable> >::const_iterator var;
459  var = variables.find(name);
460  if(var != variables.end())
461  {
462  // Step (2): We have a variable for this Oid
463 
464  // update variable
465  try
466  {
467  // Add variable to response (Step (1): include name)
468  var->second->handle_get();
469  response->varbindlist.push_back( Varbind(name, var->second) );
470  }
471  catch(...)
472  {
473  // An error occurred
474  response->set_error( ResponsePDU::genErr );
475  response->set_index( index );
476  // Leave response.varbindlist empty
477  }
478 
479  }
480  else
481  {
482  // Interpret 'name' as prefix:
483  // append .0 and check whether we have a variable
484  // with this name
485  Oid name_copy(name, 0);
486 
487  var = variables.find(name_copy);
488  if(var != variables.end())
489  {
490  // Step (4): We have a variable with the object
491  // identifier prefix 'name': Send noSuchInstance
492  // error (Step (1): include name)
493  response->varbindlist.push_back( Varbind(name, Varbind::noSuchInstance) );
494  }
495  else
496  {
497  // Step (3): we have no variable with the object
498  // identifier prefix 'name': Send noSuchObject
499  // error (Step (1): include name)
500  response->varbindlist.push_back( Varbind(name, Varbind::noSuchObject) );
501  }
502  }
503 
504  index++;
505  }
506 }

510 void MasterProxy::handle_getnextpdu(QSharedPointer<ResponsePDU> response, QSharedPointer<GetNextPDU> getnext_pdu)
511 {
512  // Handling according to
513  // RFC 2741, 7.2.3.2 "Subagent Processing of the agentx-GetNext-PDU"
514 
515  // Extract searchRange list
516  vector< pair<Oid,Oid> >& sr = getnext_pdu->get_sr();
517 
518  // Iterate over list and handle each SearchRange separately
519  vector< pair<Oid,Oid> >::const_iterator i;
520  quint16 index = 1; // Index is 1-based (RFC 2741,
521  // 5.4. "Value Representation"):
522  for(i = sr.begin(); i != sr.end(); i++)
523  {
524  // The names
525  const Oid& starting_oid = i->first;
526  const Oid& ending_oid = i->second;
527 
528  // Find "next" variable
529  map< Oid, QSharedPointer<AbstractVariable> >::const_iterator next_var;
530  if( ! starting_oid.include())
531  {
532  // Find the closest lexicographical successor to the starting
533  // OID (excluding the starting OID itself)
534  next_var = variables.upper_bound(starting_oid);
535  }
536  else
537  {
538  // Find the exact variable or, if not present, find the
539  // lexicographical successor of it
540  next_var = variables.lower_bound(starting_oid);
541  }
542  if(next_var != variables.end() && ! ending_oid.is_null() )
543  {
544  // The "next" variable must precede the ending OID (it must not
545  // be greater or equal than the ending OID)
546  if( next_var->first >= ending_oid )
547  {
548  // The found "next" variable doesn't precede the ending
549  // OID, which means that we didn't found a suitable
550  // variable.
551  next_var = variables.end(); // indicate "not found"
552  }
553  }
554 
555  if(next_var != variables.end())
556  {
557  // "Next" variable was found
558 
559  // update variable
560  try
561  {
562  next_var->second->handle_get();
563  response->varbindlist.push_back( Varbind(next_var->first, next_var->second) );
564  }
565  catch(...)
566  {
567  // An error occurred
568  response->set_error( ResponsePDU::genErr );
569  response->set_index( index );
570  // Leave response.varbindlist empty
571  }
572 
573  }
574  else
575  {
576  // "Next" variable was NOT found
577  response->varbindlist.push_back( Varbind(starting_oid, Varbind::endOfMibView) );
578  }
579 
580  index++;
581  }
582 }

586 void MasterProxy::handle_testsetpdu(QSharedPointer<ResponsePDU> response, QSharedPointer<TestSetPDU> testset_pdu)
587 {
588  // Handling according to
589  // RFC 2741, 7.2.4.1 "Subagent Processing of the agentx-TestSet-PDU"
590 
591  // Extract Varbind list
592  vector<Varbind>& vb = testset_pdu->get_vb();
593 
594  // Initially, no Varbind failed:
595  response->set_error(ResponsePDU::noAgentXError);
596 
597  // Iterate over list and handle each Varbind separately. Return on the
598  // first varbind which doesn't validate correctly.
599  vector<Varbind>::const_iterator i;
600  quint16 index;
601  for(i = vb.begin(), index = 1; i != vb.end(); i++, index++)
602  {
603  // Find the associated variable
604  map< Oid, QSharedPointer<AbstractVariable> >::const_iterator var;
605  var = variables.find(i->get_name());
606  if(var == variables.end())
607  {
608  // error: variable unknown
609  response->set_error(ResponsePDU::notWritable);
610  response->set_index(index);
611 
612  // Some variables may have allocated resources, which must be
613  // released again. This is the same as handle_cleanupsetpdu() does,
614  // so we are lazy here and call this function:
615  this->handle_cleanupsetpdu();
616  return;
617  }
618 
619  // Remember the found variable for later operations
620  setlist.push_back(var->second);
621 
622  // Perform validation, store result within response
623  // Note: ResponsePDU::error_t and variable::testset_result_t are in
624  // sync, therefore the static cast works.
625  response->set_error(static_cast<ResponsePDU::error_t>(var->second->handle_testset(i->get_var())));
626  if(response->get_error() != ResponsePDU::noAgentXError)
627  {
628  response->set_index(index);
629 
630  // Some variables may have allocated resources, which must be
631  // released again. This is the same as handle_cleanupsetpdu() does,
632  // so we are lazy here and call this function:
633  this->handle_cleanupsetpdu();
634 
635  return;
636  }
637  }
638 
639  // At this point, all VarBinds validated correctly. The response was filled
640  // correctly, and we simply return.
641  return;
642 }

730 void MasterProxy::handle_pdu(QSharedPointer<PDU> pdu)
731 {
732  int error = 0; // 0 is "success"
733  if(error == -2)
734  {
735  // Version error
736  // TODO: provide better handling. For now: ignore PDU
737  // Note: Error numbers are documented in connector.hpp
738  return;
739  }
740 
741  // Here we process all PDU's except ResponsePDU's, according to RFC 2741,
742  // 7.2.2. "Subagent Processing".
743 
744  // Step 2) Was there a parse error?
745  //
746  // Note: Error numbers are documented in connector.hpp
747  if(error == -1)
748  {
749  // Step 4b) Stop processing, don't send reply.
750  //
751  // Note: we cannot determine whether the header was sucessfully
752  // parsed, therefore we simply ignore the PDU. We don't send a
753  // parseError indication to the master agent.
754  return;
755  }
756 
757  // Step 1) Create a response and copy header from received pdu
758  //
759  // Notes:
760  // - Only if no parse error was detected
761  // - The version, type and payload_length fields are filled
762  // automatically
763  // - The flags are not copied, because they have
764  // other meanings in ResponsePDU's.
765  // - TODO: Context is not yet supported.
766  QSharedPointer<ResponsePDU> response(new ResponsePDU);
767  response->set_sessionID( pdu->get_sessionID() );
768  response->set_transactionID( pdu->get_transactionID() );
769  response->set_packetID( pdu->get_packetID() );
770  response->set_error(ResponsePDU::noAgentXError);
771  response->set_index(0);
772 
773  // Step 3) Is the session valid?
774  if(pdu->get_sessionID() != this->sessionID)
775  {
776  response->set_error(ResponsePDU::notOpen);
777

34  /**
35  * \brief Represents an SNMP object identifier (OID).
36  *
37  * This class represents an OID. OID's are sequences of sub-identifiers,
38  * which are integers.
39  *
40  * This class provides constructors taking a string which contains an OID.
41  * For example, this works:
42  *
43  * \code
44  * Oid myCompany("1.3.6.1.4.1.355");
45  * \endcode
46  *
47  * Also a constructor is provided which takes an Oid and a string and
48  * concatenates them, so this works also:
49  *
50  * \code
51  * Oid myObject(myCompany, "1.1.3.0");
52  * \endcode
53  *
54  * In addition, some common Oid's are provided as constants, e.g.
55  * 'enterprises_oid', so the following will also work:
56  *
57  * \code
58  * Oid yourCompany(enterprises_oid, 42); // 42 is an integer
59  * Oid yourCompanyMib(enterprises_oid, "42.1.1"); // "42.1.1" is a string
60  * \endcode
61  *
62  * The string given to the constructors must have a valid syntax. If a
63  * malformed string is provided, inval_param is thrown and the object is
64  * not constructed. For example, the following strings are malformed:
65  *
66  * \code
67  * "1,3,6" // wrong separator (must be a period)
68  * "1.3.6." // trailing character at the end
69  * "1.3.6.1.4.1.123456789123" // last subid is too big (must fit in a 32 bit unsigned integer)
70  * \endcode
71  *
72  * However, the following strings are accepted:
73  *
74  * \code
75  * "1.3.6"
76  * "1" // a single subid is ok
77  * "1.3.6.1.4.1.42.1.0" // 0 as subid is ok
78  * "" // empty string is ok
79  * \endcode
80  *
81  * This class inherits from QVector<quint32>, which means that an Oid
82  * object can be manipulated the same way as a QVector<> can be
83  * manipulated:
84  *
85  * \code
86  * Oid theirCompany = enterprises_oid;
87  * theirCompany.append(23); // Don't use a string here!
88  * \endcode
89  *
90  */
91  class Oid: public QVector<quint32>

169  /**
170  * \brief The less-than operator
171  *
172  * An OID is less than another OID if either the first
173  * not-identical part is lesser or if all parts are identical and
174  * it has lesser parts.
175  *
176  * Example:\n
177  * 1.3.6.1.4.1.42.3<b>.3</b>.1 \n
178  * is less than \n
179  * 1.3.6.1.4.1.42.3<b>.4</b>.1 \n
180  * Note the next to last number.
181  *
182  * Also,\n
183  * 1.3.6.1.4.1.42.3.3.1 \n
184  * is less than \n
185  * 1.3.6.1.4.1.42.3.3.1<b>.1</b> \n
186  * because it is shorter.
187  *
188  * However, \n
189  * 1.3.6.1.4.1.42.3<b>.3</b>.1 \n
190  * is greater than \n
191  * 1.3.6.1.4.1.42.3<b>.2</b>.1.1 \n
192  * because the 9th number is greater (although the first OID has
193  * less numbers than the second).
194  *
195  * \param o The OID to compare to.
196  *
197  * \return True if the OID is less than 'o', false otherwise.
198  */
199  bool operator<(const Oid& o) const;
200

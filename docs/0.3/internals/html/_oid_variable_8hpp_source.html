115  /**
116  * \brief Convert the value to an OID.
117  *
118  * The conversion is done according to RFC 2578,
119  * 7.7. "Mapping of the INDEX clause". First, the
120  * length of the OID (i.e. the number of subid's it has) is
121  * converted to a subid. Then, each subid of the value is appended.
122  *
123  * \todo Support fixed-length OIDs according to RFC 2578,
124  * 7.7. "Mapping of the INDEX clause".
125  *
126  * \return The generated Oid.
127  *
128  * \exception None.
129  */
130  virtual Oid toOid() const
131  {
132  // fixed length currently unsupported:
133  bool fixedLength = false;
134 
135  Oid oid;
136 
137  // Store length if needed
138  if(!fixedLength)
139  {
140  oid.push_back(v.size());
141  }
142 
143  // Store value
144  oid += v;
145 
146  return oid;
147  }

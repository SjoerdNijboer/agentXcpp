59 PDU::PDU(binary::const_iterator& pos,
60  const binary::const_iterator& end,
61  bool big_endian)
62 {
63  if(end - pos < 20)
64  {
65  throw(parse_error());
66  }
67 
68  // skip protocol version (already checked by get_pdu()) and
69  // type field (not needed here)
70  pos += 2;
71 
72  // read flags
73  quint8 flags = *pos++;
74  instance_registration = ( flags & (1<<0) ) ? true : false;
75  new_index = ( flags & (1<<1) ) ? true : false;
76  any_index = ( flags & (1<<2) ) ? true : false;
77  non_default_context = ( flags & (1<<3) ) ? true : false;
78

80  pos++;
81

84 
85  // read transactionID

87 
88  // read packetID
89  packetID = read32(pos, big_endian);

91  // skip payload length (not needed here)
92  pos += 4;

98 QSharedPointer<PDU> PDU::parse_pdu(binary buf)
99 {
100  // needed for parsing
101  binary::const_iterator pos;
102 
103  // check protocol version
104  quint8 version = buf[0];
105  if( version != 1 )
106  {
107  // Wrong protocol:
108  // throw exception
109  throw( version_error() );
110  }
111 
112  // read endianess flag
113  quint8 flags = buf[2];
114  bool big_endian = ( flags & (1<<4) ) ? true : false;
115 
116  // read payload length
117  quint32 payload_length;
118  pos = buf.begin() + 16;
119  payload_length = read32(pos, big_endian);
120  if( payload_length % 4 != 0 )
121  {
122  // payload length must be a multiple of 4!
123  // See RFC 2741, 6.1. "AgentX PDU Header"
124  // -> throw exception
125  throw( parse_error() );
126  }
127 
128  // read PDU type
129  quint8 type = buf[1];
130 
131  // create PDU (TODO: complete the list!)
132  QSharedPointer<PDU> pdu;
133  pos = buf.begin();
134  const binary::const_iterator end = buf.end();
135  switch(type)
136  {
137  case agentxOpenPDU:
138  pdu = QSharedPointer<PDU>(new OpenPDU(pos, end, big_endian));
139  break;
140  case agentxClosePDU:
141  pdu = QSharedPointer<PDU>(new ClosePDU(pos, end, big_endian));
142  break;
143  case agentxRegisterPDU:
144  pdu = QSharedPointer<PDU>(new RegisterPDU(pos, end, big_endian));
145  break;
146  case agentxUnregisterPDU:
147  pdu = QSharedPointer<PDU>(new UnregisterPDU(pos, end, big_endian));
148  break;
149  case agentxResponsePDU:
150  pdu = QSharedPointer<PDU>(new ResponsePDU(pos, end, big_endian));
151  break;
152  case agentxCommitSetPDU:
153  pdu = QSharedPointer<PDU>(new CommitSetPDU(pos, end, big_endian));
154  break;
155  case agentxUndoSetPDU:
156  pdu = QSharedPointer<PDU>(new UndoSetPDU(pos, end, big_endian));
157  break;
158  case agentxTestSetPDU:
159  pdu = QSharedPointer<PDU>(new TestSetPDU(pos, end, big_endian));
160  break;
161  case agentxCleanupSetPDU:
162  pdu = QSharedPointer<PDU>(new CleanupSetPDU(pos, end, big_endian));
163  break;
164  case agentxGetPDU:
165  pdu = QSharedPointer<PDU>(new GetPDU(pos, end, big_endian));
166  break;
167  case agentxGetNextPDU:
168  pdu = QSharedPointer<PDU>(new GetNextPDU(pos, end, big_endian));
169  break;
170  case agentxGetBulkPDU:
171  pdu = QSharedPointer<PDU>(new GetBulkPDU(pos, end, big_endian));
172  break;
173  case agentxNotifyPDU:
174  pdu = QSharedPointer<PDU>(new NotifyPDU(pos, end, big_endian));
175  break;
176  default:
177  // type is invalid
178  throw(parse_error());
179  }
180 
181  // return created PDU
182  return pdu;
183 }

187 void PDU::add_header(type_t type, binary& payload) const
188 {
189  /* Construct header */
190  binary header;
191 
192  // Protocol version
193  header.push_back(1);
194 
195  // Type
196  header.push_back(type);
197 
198  // flags
199  quint8 flags = 0;
200  if(instance_registration) flags |= (1<<0);
201  if(new_index) flags |= (1<<1);
202  if(any_index) flags |= (1<<2);
203  if(non_default_context) flags |= (1<<3);
204  flags |= (1<<4); // We always use big endian
205  header.push_back(flags);
206 
207  // reserved field
208  header.push_back(0);
209 
210  // remaining fields
211  write32(header, sessionID);
212  write32(header, transactionID);
213  write32(header, packetID);
214  write32(header, payload.size()); // payload length
215 
216  // Add header to payload
217  payload.insert(0, header);
218 }

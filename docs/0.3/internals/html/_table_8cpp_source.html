32  if(! myMasterProxy)

37  // Ensure that index is not registered
38  if(entries.contains(entry))
39  {
40  // The entry was already added

43 
44  // Calculate entry's index
45  Oid entryIndex;
46  QVector< QSharedPointer<AbstractVariable> > indexVariables = entry->indexVariables();
47  QVectorIterator< QSharedPointer<AbstractVariable> > iter(indexVariables);
48  while(iter.hasNext())
49  {
50  QSharedPointer<AbstractVariable> var = iter.next();
51  if(!var)
52  {
53  // No variable -> fail
54  return false;
55  }
56  Oid variableIndex = var->toOid();
57  if(variableIndex.is_null())
58  {
59  // Variable cannot be converted to Oid -> not
60  // allowed as index variable.
61  return false;
62  }
63  else
64  {
65  // Variable contributes to index
66  entryIndex += variableIndex;
67  }
68  }
69 
70  // Register entry
71  entries[entry] = entryIndex;
72 
73  // Register all variables of the entry with the MasterProxy object
74  QMap< quint32, QSharedPointer<AbstractVariable> > variables = entry->variables();
75  QMapIterator< quint32, QSharedPointer<AbstractVariable> > iter2(variables);
76  QVector< QPair< Oid,QSharedPointer<AbstractVariable> > > toRegister;
77  while(iter2.hasNext())

79  iter2.next();
80  if(! iter2.value())
81  {
82  // No variable given -> fail

87  toRegister.append(qMakePair(myOid + entry->subid + iter2.key() + entryIndex, iter2.value()));
88  }

96 bool Table::removeEntry(QSharedPointer<TableEntry> entry)
97 {
98  // Check for MasterProxy object
99  if(! myMasterProxy)
100  {
101  return false;
102  }
103 
104  // Ensure that entry is known
105  if(!entries.contains(entry))
106  {
107  // Entry not found
108  return false;
109  }
110 
111  // Unregister all variables of the entry
112  Oid entryIndex = entries[entry]; // Use index at time of registration
113  QMap< quint32, QSharedPointer<AbstractVariable> > variables = entry->variables();
114  QMapIterator< quint32, QSharedPointer<AbstractVariable> > iter2(variables);
115  QVector<Oid> toUnregister;
116  while(iter2.hasNext())
117  {
118  iter2.next();
119  // Add variable's Oid to list
120  toUnregister.append(myOid + entry->subid + iter2.key() + entryIndex);
121  }
122  myMasterProxy->removeVariables(toUnregister);
123 
124  // Remove entry from local storage
125  entries.remove(entry);
126

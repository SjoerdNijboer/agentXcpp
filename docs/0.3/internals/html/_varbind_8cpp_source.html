100 Varbind::Varbind(binary::const_iterator& pos,
101  const binary::const_iterator& end,
102  bool big_endian)
103 {
104  quint16 type;
105 
106  // Type and reserved field
107  if(end - pos < 4)
108  {
109  throw(parse_error());
110  }
111 
112  // Get type
113  type = read16(pos, big_endian);
114 
115  // skip reserved field
116  pos += 2;
117 
118  // read OID: no exceptions are catched; they are forwarded to the caller
119  name = OidVariable(pos, end, big_endian).value();
120 
121  // Get data: no exceptions are catched; they are forwarded to the caller
122  switch(type)

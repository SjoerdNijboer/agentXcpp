00059 varbind NotifyPDU::trapToNotification(oid enterprise,

00061                                       uint32_t specific_trap)
00062 {
00063     // The snmpTrapOID type
00064     //
00065     // According to RFC 1907:
00066     // snmpTrapOID ::= snmpTrap.1
00067     // snmpTrap ::= snmpMIBObjects.4
00068     // snmpMIBObjects ::= snmpMIB.1
00069     // snmpMIB ::= snmpModules.1
00070     //
00071     // According to RFC 2578 (SNMPv2-SMI):
00072     // snmpModules ::= snmpV2.3
00073     // snmpV2 ::= internet.6
00074     // 
00075     // conclusion:
00076     static const oid snmpMIBObjects(internet,6,3,1,1);
00077     static const oid snmpTrapOID(snmpMIBObjects,4,1);
00078 
00079 
00080     // We need the OID of the SNMPv1 traps. These are defined here.
00081     //
00082     // First we define a "helper" OID:
00083     static const oid snmpTraps(snmpMIBObjects,5);
00084     //
00085     // Some traps according to RFC 1907:
00086     static const oid snmpTraps_coldStart(snmpTraps,1);
00087     static const oid snmpTraps_warmStart(snmpTraps,2);
00088     static const oid snmpTraps_authenticationFailure(snmpTraps,5);
00089     //
00090     // Some traps according to RFC 1573:
00091     static const oid snmpTraps_linkDown(snmpTraps,3);
00092     static const oid snmpTraps_linkUp(snmpTraps,4);
00093     //
00094     // Finally, egpNeighborLoss. According to RC 1907 it is defined in RFC 
00095     // 1213, however, the latter doesn't in fact define it. On the other hand, 
00096     // RFC 2089 defines egpNeighborLoss as 1.3.6.1.6.3.1.1.5.6, which is 
00097     // snmpTraps.6 and corresponds to the comment in RFC 1907, so we use this 
00098     // one:
00099     static const oid snmpTraps_egpNeighborLoss(snmpTraps,6);
00100 
00101 
00102     // calculate the value of snmpTrapOID.0 according to RFC 1908
00103     oid* value;
00104 
00105     switch(generic_trap)
00106     {
00107         case coldStart:
00108             value = new oid(coldStart);
00109             break;
00110         case warmStart:
00111             value = new oid(warmStart);
00112             break;
00113         case linkDown:
00114             value = new oid(linkDown);
00115             break;
00116         case linkUp:
00117             value = new oid(linkUp);
00118             break;
00119         case authenticationFailure:
00120             value = new oid(authenticationFailure);
00121             break;
00122         case egpNeighborLoss:
00123             value = new oid(egpNeighborLoss);
00124             break;
00125         case enterpriseSpecific:
00126             value = new oid(enterprise, 0, specific_trap);
00127             break;
00128         default:
00129             // invalid generic_trap value!
00130             throw(inval_param());
00131     }
00132 
00133     // Create and return varbind
00134     return varbind( new oid(snmpTrapOID,0), value);

00026 Octet_String::Octet_String(std::string v)
00027 {
00028     // Here we convert initial value to a data_t string. We do this in three 
00029     // steps:
00030     // 1. get the bare data: v.data()
00031     // 2. cast the data to the value type of data_t (i.e. byte_t)
00032     // 3. calculate the size of the data
00033     //    - v.size() gives us the number of characters
00034     //    - sizeof(std::string::value_type) gives us the size of an character
00035     // 4. Assign v to value, giving it the bare data and its size
00036     //
00037     // This seems to be goofy, but it ensures that our code works even on a 
00038     // machine where a char is not 8 bit wide, i.e. when char has another size 
00039     // than byte_t.
00040     value.assign(
00041             reinterpret_cast<const data_t::value_type*>( v.data() ),
00042             v.size() * sizeof( std::string::value_type )
00043                 );
00044 }

00046 data_t Octet_String::serialize() const
00047 {
00048     data_t serialized;
00049 
00050     // encode size (big endian)
00051     int size = value.size();
00052     serialized.push_back(size >> 24 & 0xff);
00053     serialized.push_back(size >> 16 & 0xff);
00054     serialized.push_back(size >> 8 & 0xff);
00055     serialized.push_back(size >> 0 & 0xff);
00056 
00057     // encode value
00058     serialized += value;
00059 
00060     // Padding bytes
00061     int padsize = 4 - (size % 4);
00062     if( padsize == 4 ) padsize = 0; // avoid adding 4 padding bytes
00063     while( padsize-- )
00064     {
00065         serialized.push_back(0);
00066     }
00067 
00068     return serialized;

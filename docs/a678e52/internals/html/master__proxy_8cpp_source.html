00112 void master_proxy::connect()
00113 {
00114     if( this->connection->is_connected() )
00115     {
00116         // we are already connected -> nothing to do

00118     }
00119 
00120     // Connect to endpoint
00121     try
00122     {
00123         this->connection->connect();
00124     }
00125     catch(boost::system::system_error)
00126     {
00127         throw;
00128     }
00129     // The response we expect from the master
00130     boost::shared_ptr<ResponsePDU> response;
00131 
00132     try
00133     {
00134         // Send OpenPDU
00135         OpenPDU openpdu;
00136         openpdu.set_timeout(default_timeout);
00137         openpdu.set_id(id);
00138         // throws disconnected and timeout_error:
00139         this->connection->send(openpdu);
00140 
00141         // Wait for response
00142         // throws disconnected and timeout_error:
00143         response = this->connection->wait_for_response(openpdu.get_packetID());
00144     }
00145     catch(disconnected)
00146     {
00147         throw; // forward
00148     }
00149     catch(timeout_error)
00150     {
00151         throw disconnected();
00152     }
00153 
00154     // Check for errors
00155     if(response->get_error() != ResponsePDU::noAgentXError)
00156     {
00157         // Some error occured, disconnect
00158         this->connection->disconnect();
00159         throw disconnected();
00160     }
00161 
00162     // All went fine, we are connected now
00163     this->sessionID = response->get_sessionID();

00169 void master_proxy::disconnect(ClosePDU::reason_t reason)
00170 {
00171     if( ! this->connection->is_connected() )
00172     {
00173         // we are already disconnected -> nothing to do
00174         return;
00175     }
00176 
00177     // According to RFC 2741, 7.1.8. "Processing the agentx-Close-PDU", the 
00178     // master agent unregisters all MIB regions, frees all index values and all 
00179     // sysORID are removed. Thus no need to clean up before ClosePDU is sent.
00180     
00181     // The response we expect from the master
00182     boost::shared_ptr<ResponsePDU> response;
00183 
00184     // Try clean shutdown
00185     try
00186     {
00187         // Send ClosePDU
00188         ClosePDU closepdu(this->sessionID, reason);
00189         // throws disconnected and timeout_error:
00190         this->connection->send(closepdu);
00191         
00192         // Wait for response
00193         // throws disconnected and timeout_error:
00194         response = this->connection->wait_for_response(closepdu.get_packetID());
00195     
00196         // Check for errors
00197         if(response->get_error() != ResponsePDU::noAgentXError)
00198         {
00199             // Some error occured. However, We will soon disconnect anyway.
00200             // -> ignore
00201             // TODO: Insert log message here.
00202         }
00203     }
00204     catch(...)
00205     {
00206         // We will soon disconnect anyway.
00207         // -> ignore all errors
00208     }
00209 
00210     // Finally: disconnect
00211     this->connection->disconnect();

00256         // Throw timeout exception

00259 
00260     // Check Response
00261     switch(response->get_error())
00262     {
00263         // General errors:
00264 
00265         case ResponsePDU::parseError:
00266             // Oops, we sent a malformed PDU to the master
00267             throw(internal_error());
00268 
00269         case ResponsePDU::notOpen:
00270             // We checked the connection state before, but maybe we lost the 
00271             // connection during communication...
00272             throw(disconnected());
00273 
00274         case ResponsePDU::unsupportedContext:
00275             // We do currently not really support contexts in this library. An 
00276             // invalid context is thus probably an agentxcpp bug.
00277             throw(internal_error());
00278 
00279         case ResponsePDU::processingError:
00280             // master was unable to process the request
00281             throw(master_is_unable());
00282 
00283         case ResponsePDU::noAgentXError:
00284             // Hey, it worked! Nothing left to do here, continuing...

00288
